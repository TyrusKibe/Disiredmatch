Function retuser
Parameters fakestr
m.thisstr = Alltrim(fakestr)
N = 1
actstr = ""
Do While N <= Len(m.thisstr)
	i = Substr(m.thisstr,N,1)
	If !Empty(i)
		ival = Asc(i)-N-Len(m.thisstr)
		actstr = actstr+Chr(ival)
	Endif
	N = N+1
Enddo
Return actstr
